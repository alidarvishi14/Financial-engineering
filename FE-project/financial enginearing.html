import matplotlib.pyplot as plt

def bionomial_grid(n,stockValues,optionValues):
    high = 8
    width =12
    plt.figure(figsize=(width,high))
    G=nx.Graph()
    for i in range(0,n+1):
        for j in range(1,i+2):
            plt.figtext(0.03*np.exp(-i)+i/n,0.05*np.exp(-i)+
                        0.05+0.5-0.5*(i/2-j+1)/n*2,"Stock price="+str(stockValues[i,i-(j-1)]))
            plt.figtext(0.03*np.exp(-i)+i/n,0.05*np.exp(-i)+
                        0.5-0.5*(i/2-j+1)/n*2,"option price="+str(optionValues[i,i-(j-1)]))
            if i<n:
                G.add_edge((i,j),(i+1,j))
                G.add_edge((i,j),(i+1,j+1))
    posG={}
    for node in G.nodes():
        posG[node]=(node[0],n+2+node[0]-2*node[1])
    nx.draw(G,pos=posG)

import pandas as pd
import scipy.stats as si
count = 10
N_sim = 1000
days = 250
S_0 = np.round(np.random.normal(95,10,count),3)
K_1 = np.round(np.random.uniform(80,90,count),0)
K_2 = np.round(np.random.uniform(90,100,count),0)
T = 1
volatility = 0.25
riskFreeRate = 0.12
data = pd.DataFrame(columns=['S0','K1','K2','Gap MC','Standard MC','Standard BSM','col 5-col 4','Theory'])
for i in range(count):
    prices = S_0[i] * np.exp( np.cumsum( (riskFreeRate - 0.5 * volatility ** 2) *
                                        T/days + volatility * np.sqrt(T/days) *
                                        np.random.standard_normal((days + 1, N_sim)), axis=0))
#     prices = np.prod(1+returns,1)*S_0[i]
    gapCall = round(np.exp(-riskFreeRate*T)*np.mean((prices[-1,:]-K_2[i])*(prices[-1,:]>K_1[i])),3)
    standardCall = round(np.exp(-riskFreeRate*T)*np.mean(np.maximum(prices[-1,:]-K_1[i],0)),3)
    d1 = (np.log(S_0[i] / K_1[i]) + (riskFreeRate + 0.5 * volatility ** 2) * T) / (volatility * np.sqrt(T))
    d2 = (np.log(S_0[i] / K_1[i]) + (riskFreeRate - 0.5 * volatility ** 2) * T) / (volatility * np.sqrt(T))
    call = (S_0[i] * si.norm.cdf(d1, 0.0, 1.0) - K_1[i] * np.exp(-riskFreeRate * T) * si.norm.cdf(d2, 0.0, 1.0))
    BScall = round(call,3)
    KS = (K_2[i]-K_1[i])*np.exp(-riskFreeRate*T)*si.norm.cdf(d2, 0.0, 1.0)
    KS = round(KS,3)
    data = data.append({'S0': S_0[i],'K1':K_1[i],'K2':K_2[i],
                        'Gap MC':gapCall,'Standard MC':standardCall,'Standard BSM':BScall,
                        'col 5-col 4':-gapCall+standardCall,'Theory':KS}, ignore_index=True,)
